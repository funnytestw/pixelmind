# استخدام نسخة PHP مع Apache جاهزة للارافيل
FROM php:8.2-apache

# تثبيت الإضافات اللازمة للارافيل وقاعدة البيانات
RUN apt-get update && apt-get install -y \
    libpng-dev \
    libonig-dev \
    libxml2-dev \
    zip \
    unzip \
    git \
    curl

# تثبيت إضافات PHP
RUN docker-php-ext-install pdo_mysql mbstring exif pcntl bcmath gd

# تفعيل Apache Rewrite Module (ضروري جداً للارافيل)
RUN a2enmod rewrite

# ضبط مجلد العمل داخل السيرفر
WORKDIR /var/www/html

# نسخ ملفات المشروع إلى السيرفر
COPY . .

# تثبيت Composer
COPY --from=composer:latest /usr/bin/composer /usr/bin/composer
RUN composer install --no-dev --optimize-autoloader

# تغيير ملكية الملفات لـ Apache
RUN chown -R www-data:www-data /var/www/html/storage /var/www/html/bootstrap/cache

# تعديل إعدادات Apache ليعمل من مجلد public
ENV APACHE_DOCUMENT_ROOT /var/www/html/public
RUN sed -ri -e 's!/var/www/html!${APACHE_DOCUMENT_ROOT}!g' /etc/apache2/sites-available/*.conf
RUN sed -ri -e 's!/var/www/html!${APACHE_DOCUMENT_ROOT}!g' /etc/apache2/apache2.conf /etc/apache2/conf-available/*.conf

# فتح المنفذ 80
EXPOSE 80

CMD ["apache2-foreground"]
